104 // The main purpose of Calculate() is to determine what phase the engine should
105 // be in, then call the corresponding function.
106 
107 void FGTurbine::Calculate(void)
108 {
109  double thrust;
110 
111  RunPreFunctions();
112 
113  ThrottlePos = in.ThrottlePos[EngineNumber];
114 
115  if (ThrottlePos > 1.0) {
116  AugmentCmd = ThrottlePos - 1.0;
117  ThrottlePos -= AugmentCmd;
118  } else {
119  AugmentCmd = 0.0;
120  }
121 
122  // When trimming is finished check if user wants engine OFF or RUNNING
123  if ((phase == tpTrim) && (in.TotalDeltaT > 0)) {
124  if (Running && !Starved) {
125  phase = tpRun;
126  N1_factor = MaxN1 - IdleN1;
127  N2_factor = MaxN2 - IdleN2;
128  N2 = IdleN2 + ThrottlePos * N2_factor;
129  N1 = IdleN1 + ThrottlePos * N1_factor;
130  OilTemp_degK = 366.0;
131  Cutoff = false;
132  } else {
133  phase = tpOff;
134  Cutoff = true;
135  EGT_degC = in.TAT_c;
136  }
137  }
138 
139  if (!Running && Cutoff && Starter) {
140  if (phase == tpOff) phase = tpSpinUp;
141  }
142 
143  // start
144  if ((Starter == true) || (in.qbar > 30.0)) {
145  if (!Running && !Cutoff && (N2 > 15.0)) phase = tpStart;
146  }
147 
148  if (Cutoff && (phase != tpSpinUp)) phase = tpOff;
149  if (in.TotalDeltaT == 0) phase = tpTrim;
150  if (Starved) phase = tpOff;
151  if (Stalled) phase = tpStall;
152  if (Seized) phase = tpSeize;
153 
154  switch (phase) {
155  case tpOff: thrust = Off(); break;
156  case tpRun: thrust = Run(); break;
157  case tpSpinUp: thrust = SpinUp(); break;
158  case tpStart: thrust = Start(); break;
159  case tpStall: thrust = Stall(); break;
160  case tpSeize: thrust = Seize(); break;
161  case tpTrim: thrust = Trim(); break;
162  default: thrust = Off();
163  }
164 
165  Thruster->Calculate(thrust); // allow thruster to modify thrust (i.e. reversing)
166 
167  RunPostFunctions();
168 }

172 double FGTurbine::Off(void)
173 {
174  Running = false;
175  FuelFlow_pph = Seek(&FuelFlow_pph, 0, 1000.0, 10000.0);
176  // some engines have inlets that close when they are off. So, if a flag is true disable windmilling
177  if (disableWindmill == false) {
178  // Need a small non-zero increment for acceleration otherwise acceleration will be 0 if N1 = 0
179  N1 = Seek(&N1, in.qbar/10.0, N1/2.0 + 0.1, N1/N1_spindown);
180  N2 = Seek(&N2, in.qbar/15.0, N2/2.0 + 0.1, N2/N2_spindown);
181  } else {
182  N1 = Seek(&N1, 0, N1/2.0, N1/N1_spindown);
183  N2 = Seek(&N2, 0, N2/2.0, N2/N2_spindown);
184  }
185  EGT_degC = Seek(&EGT_degC, in.TAT_c, 11.7, 7.3);
186  OilTemp_degK = Seek(&OilTemp_degK, in.TAT_c + 273.0, 0.2, 0.2);
187  OilPressure_psi = N2 * 0.62;
188  NozzlePosition = Seek(&NozzlePosition, 1.0, 0.8, 0.8);
189  EPR = Seek(&EPR, 1.0, 0.2, 0.2);
190  Augmentation = false;
191  return 0.0;
192 }

196 double FGTurbine::Run()
197 {
198  double idlethrust, milthrust, thrust;
199 
200  idlethrust = MilThrust * IdleThrustLookup->GetValue();
201  milthrust = (MilThrust - idlethrust) * MilThrustLookup->GetValue();
202 
203  Running = true;
204  Starter = false;
205 
206  N1_factor = MaxN1 - IdleN1;
207  N2_factor = MaxN2 - IdleN2;
208  if ((Injected == 1) && Injection && (InjWaterNorm > 0)) {
209  N1_factor += InjN1increment;
210  N2_factor += InjN2increment;
211  }
212  N2 = Seek(&N2, IdleN2 + ThrottlePos * N2_factor,
213  N2SpoolUp->GetValue(), N2SpoolDown->GetValue());
214  N1 = Seek(&N1, IdleN1 + ThrottlePos * N1_factor,
215  N1SpoolUp->GetValue(), N1SpoolDown->GetValue());
216  N2norm = (N2 - IdleN2) / N2_factor;
217  thrust = idlethrust + (milthrust * N2norm * N2norm);
218  EGT_degC = in.TAT_c + 363.1 + ThrottlePos * 357.1;
219  OilPressure_psi = N2 * 0.62;
220  OilTemp_degK = Seek(&OilTemp_degK, 366.0, 1.2, 0.1);
221 
222  if (!Augmentation) {
223  correctedTSFC = TSFC->GetValue();
224  FuelFlow_pph = Seek(&FuelFlow_pph, thrust * correctedTSFC, 1000.0, 10000.0);
225  if (FuelFlow_pph < IdleFF) FuelFlow_pph = IdleFF;
226  NozzlePosition = Seek(&NozzlePosition, 1.0 - N2norm, 0.8, 0.8);
227  thrust = thrust * (1.0 - BleedDemand);
228  EPR = 1.0 + thrust/MilThrust;
229  }
230 
231  if (AugMethod == 1) {
232  if ((ThrottlePos > 0.99) && (N2 > 97.0)) {Augmentation = true;}
233  else {Augmentation = false;}
234  }
235 
236  if ((Augmented == 1) && Augmentation && (AugMethod < 2)) {
237  thrust = MaxThrustLookup->GetValue() * MaxThrust;
238  FuelFlow_pph = Seek(&FuelFlow_pph, thrust * ATSFC->GetValue(), 5000.0, 10000.0);
239  NozzlePosition = Seek(&NozzlePosition, 1.0, 0.8, 0.8);
240  }
241 
242  if (AugMethod == 2) {
243  if (AugmentCmd > 0.0) {
244  Augmentation = true;
245  double tdiff = (MaxThrust * MaxThrustLookup->GetValue()) - thrust;
246  thrust += (tdiff * AugmentCmd);
247  FuelFlow_pph = Seek(&FuelFlow_pph, thrust * ATSFC->GetValue(), 5000.0, 10000.0);
248  NozzlePosition = Seek(&NozzlePosition, 1.0, 0.8, 0.8);
249  } else {
250  Augmentation = false;
251  }
252  }
253 
254  if ((Injected == 1) && Injection && (InjWaterNorm > 0.0)) {
255  InjectionTimer += in.TotalDeltaT;
256  if (InjectionTimer < InjectionTime) {
257  thrust = thrust * InjectionLookup->GetValue();
258  InjWaterNorm = 1.0 - (InjectionTimer/InjectionTime);
259  } else {
260  Injection = false;
261  InjWaterNorm = 0.0;
262  }
263  }
264 
265  if (Cutoff) phase = tpOff;
266  if (Starved) phase = tpOff;
267 
268  return thrust;
269 }

273 double FGTurbine::SpinUp(void)
274 {
275  Running = false;
276  FuelFlow_pph = 0.0;
277  N2 = Seek(&N2, IgnitionN2, N2_spinup, N2/2.0);
278  N1 = Seek(&N1, IgnitionN1, N1_spinup, N1/2.0);
279  EGT_degC = Seek(&EGT_degC, in.TAT_c, 11.7, 7.3);
280  OilPressure_psi = N2 * 0.62;
281  OilTemp_degK = Seek(&OilTemp_degK, in.TAT_c + 273.0, 0.2, 0.2);
282  EPR = 1.0;
283  NozzlePosition = 1.0;
284  if (Starter == false) phase = tpOff;
285  return 0.0;
286 }

290 double FGTurbine::Start(void)
291 {
292  if ((N2 > 15.0) && !Starved) { // minimum 15% N2 needed for start
293  Cranking = true; // provided for sound effects signal
294  if (N2 < IdleN2) {
295  N2 = Seek(&N2, IdleN2, N2_start_rate, N2/2.0);
296  N1 = Seek(&N1, IdleN1, N1_start_rate, N1/2.0);
297  EGT_degC = Seek(&EGT_degC, in.TAT_c + 363.1, 21.3, 7.3);
298  FuelFlow_pph = IdleFF * N2 / IdleN2;
299  OilPressure_psi = N2 * 0.62;
300  if ((Starter == false) && (in.qbar < 30.0)) phase = tpOff; // aborted start
301  }
302  else {
303  phase = tpRun;
304  Running = true;
305  Starter = false;
306  Cranking = false;
307  }
308  }
309  else { // no start if N2 < 15%
310  phase = tpOff;
311  Starter = false;
312  }
313 
314  return 0.0;
315 }

319 double FGTurbine::Stall(void)
320 {
321  EGT_degC = in.TAT_c + 903.14;
322  FuelFlow_pph = IdleFF;
323  N1 = Seek(&N1, in.qbar/10.0, 0, N1/10.0);
324  N2 = Seek(&N2, in.qbar/15.0, 0, N2/10.0);
325  if (ThrottlePos < 0.01) {
326  phase = tpRun; // clear the stall with throttle to idle
327  Stalled = false;
328  }
329  return 0.0;
330 }

334 double FGTurbine::Seize(void)
335 {
336  N2 = 0.0;
337  N1 = Seek(&N1, in.qbar/20.0, 0, N1/15.0);
338  FuelFlow_pph = Cutoff ? 0.0 : IdleFF;
339  OilPressure_psi = 0.0;
340  OilTemp_degK = Seek(&OilTemp_degK, in.TAT_c + 273.0, 0, 0.2);
341  Running = false;
342  return 0.0;

347 double FGTurbine::Trim()
348 {
349  double idlethrust = MilThrust * IdleThrustLookup->GetValue();
350  double milthrust = (MilThrust - idlethrust) * MilThrustLookup->GetValue();
351  double N2 = IdleN2 + ThrottlePos * N2_factor;
352  double N2norm = (N2 - IdleN2) / N2_factor;
353  double thrust = (idlethrust + (milthrust * N2norm * N2norm))
354  * (1.0 - BleedDemand);
355 
356  if (AugMethod == 1) {
357  if ((ThrottlePos > 0.99) && (N2 > 97.0)) {Augmentation = true;}
358  else {Augmentation = false;}
359  }
360 
361  if ((Augmented == 1) && Augmentation && (AugMethod < 2)) {
362  thrust = MaxThrust * MaxThrustLookup->GetValue();
363  }
364 
365  if (AugMethod == 2) {
366  if (AugmentCmd > 0.0) {
367  double tdiff = (MaxThrust * MaxThrustLookup->GetValue()) - thrust;
368  thrust += (tdiff * AugmentCmd);
369  }
370  }
371 
372  if ((Injected == 1) && Injection) {
373  thrust = thrust * InjectionLookup->GetValue();
374  }
375 
376  return thrust;
377 }

400 double FGTurbine::Seek(double *var, double target, double accel, double decel) {
401  double v = *var;
402  if (v > target) {
403  v -= in.TotalDeltaT * decel;
404  if (v < target) v = target;
405  } else if (v < target) {
406  v += in.TotalDeltaT * accel;
407  if (v > target) v = target;
408  }
409  return v;
410 }

414 bool FGTurbine::Load(FGFDMExec* exec, Element *el)
415 {
416  Element* function_element = el->FindElement("function");
417 
418  while(function_element) {
419  string name = function_element->GetAttributeValue("name");
420  if (name == "IdleThrust" || name == "MilThrust" || name == "AugThrust"
421  || name == "Injection" || name == "N1SpoolUp" || name == "N1SpoolDown"
422  || name == "N2SpoolUp" || name == "N2SpoolDown")
423  function_element->SetAttributeValue("name", string("propulsion/engine[#]/") + name);
424 
425  function_element = el->FindNextElement("function");
426  }
427 
428  FGEngine::Load(exec, el);
429 
430  ResetToIC();
431 
432  if (el->FindElement("milthrust"))
433  MilThrust = el->FindElementValueAsNumberConvertTo("milthrust","LBS");
434  if (el->FindElement("maxthrust"))
435  MaxThrust = el->FindElementValueAsNumberConvertTo("maxthrust","LBS");
436  if (el->FindElement("bypassratio"))
437  BypassRatio = el->FindElementValueAsNumber("bypassratio");
438  if (el->FindElement("bleed"))
439  BleedDemand = el->FindElementValueAsNumber("bleed");
440  if (el->FindElement("ignitionn1"))
441  IgnitionN1 = el->FindElementValueAsNumber("ignitionn1");
442  if (el->FindElement("ignitionn2"))
443  IgnitionN2 = el->FindElementValueAsNumber("ignitionn2");
444  if (el->FindElement("idlen1"))
445  IdleN1 = el->FindElementValueAsNumber("idlen1");
446  if (el->FindElement("idlen2"))
447  IdleN2 = el->FindElementValueAsNumber("idlen2");
448  if (el->FindElement("maxn1"))
449  MaxN1 = el->FindElementValueAsNumber("maxn1");
450  if (el->FindElement("maxn2"))
451  MaxN2 = el->FindElementValueAsNumber("maxn2");
452  if (el->FindElement("n1spinup"))
453  N1_spinup = el->FindElementValueAsNumber("n1spinup");
454  if (el->FindElement("n2spinup"))
455  N2_spinup = el->FindElementValueAsNumber("n2spinup");
456  if (el->FindElement("n1startrate"))
457  N1_start_rate = el->FindElementValueAsNumber("n1startrate");
458  if (el->FindElement("n2startrate"))
459  N2_start_rate = el->FindElementValueAsNumber("n2startrate");
460  if (el->FindElement("n1spindown"))
461  N1_spindown = el->FindElementValueAsNumber("n1spindown");
462  if (el->FindElement("n2spindown"))
463  N2_spindown = el->FindElementValueAsNumber("n2spindown");
464  if (el->FindElement("augmented"))
465  Augmented = (int)el->FindElementValueAsNumber("augmented");
466  if (el->FindElement("augmethod"))
467  AugMethod = (int)el->FindElementValueAsNumber("augmethod");
468  if (el->FindElement("injected"))
469  Injected = (int)el->FindElementValueAsNumber("injected");
470  if (el->FindElement("injection-time")){
471  InjectionTime = el->FindElementValueAsNumber("injection-time");
472  InjWaterNorm =1.0;
473  }
474  if (el->FindElement("injection-N1-inc"))
475  InjN1increment = el->FindElementValueAsNumber("injection-N1-inc");
476  if (el->FindElement("injection-N2-inc"))
477  InjN2increment = el->FindElementValueAsNumber("injection-N2-inc");
478  if (el->FindElement("disable-windmill"))
479  disableWindmill = el->FindElementValueAsBoolean("disable-windmill");
480  string property_prefix = CreateIndexedPropertyName("propulsion/engine", EngineNumber);
481 
482  IdleThrustLookup = GetPreFunction(property_prefix+"/IdleThrust");
483  MilThrustLookup = GetPreFunction(property_prefix+"/MilThrust");
484  MaxThrustLookup = GetPreFunction(property_prefix+"/AugThrust");
485  InjectionLookup = GetPreFunction(property_prefix+"/Injection");
486 
487  JSBSim::Element* tsfcElement = el->FindElement("tsfc");
488  if (tsfcElement) {
489  string value = tsfcElement->GetDataLine();
490  if (is_number(trim(value)))
491  TSFC = std::make_unique<FGSimplifiedTSFC>(this, atof_locale_c(value));
492  else
493  TSFC = std::make_unique<FGFunction>(FDMExec, tsfcElement, to_string(EngineNumber));
494  }
495 
496  JSBSim::Element* atsfcElement = el->FindElement("atsfc");
497  if (atsfcElement) {
498  string value = atsfcElement->GetDataLine();
499  if (is_number(trim(value)))
500  ATSFC = std::make_unique<FGRealValue>(atof_locale_c(value));
501  else
502  ATSFC = std::make_unique<FGFunction>(FDMExec, atsfcElement, to_string((int)EngineNumber));
503  }
504 
505  // Pre-calculations and initializations
506  N1SpoolUp = GetPreFunction(property_prefix+"/N1SpoolUp");
507  if (!N1SpoolUp)
508  N1SpoolUp = std::make_shared<FGSpoolUp>(this, BypassRatio, 1.0);
509 
510  N1SpoolDown = GetPreFunction(property_prefix+"/N1SpoolDown");
511  if (!N1SpoolDown)
512  N1SpoolDown = std::make_shared<FGSpoolUp>(this, BypassRatio, 2.4);
513 
514  N2SpoolUp = GetPreFunction(property_prefix+"/N2SpoolUp");
515  if (!N2SpoolUp)
516  N2SpoolUp = std::make_shared<FGSpoolUp>(this, BypassRatio, 1.0);
517 
518  N2SpoolDown = GetPreFunction(property_prefix+"/N2SpoolDown");
519  if (!N2SpoolDown)
520  N2SpoolDown = std::make_shared<FGSpoolUp>(this, BypassRatio, 3.0);
521 
522  N1_factor = MaxN1 - IdleN1;
523  N2_factor = MaxN2 - IdleN2;
524  OilTemp_degK = in.TAT_c + 273.0;
525  IdleFF = pow(MilThrust, 0.2) * 107.0; // just an estimate
526 
527  bindmodel(exec->GetPropertyManager().get());
528  return true;
529 }

559 void FGTurbine::bindmodel(FGPropertyManager* PropertyManager)
560 {
561  string property_name, base_property_name;
562  base_property_name = CreateIndexedPropertyName("propulsion/engine", EngineNumber);
563  property_name = base_property_name + "/n1";
564  PropertyManager->Tie( property_name.c_str(), &N1);
565  property_name = base_property_name + "/n2";
566  PropertyManager->Tie( property_name.c_str(), &N2);
567  property_name = base_property_name + "/injection_cmd";
568  PropertyManager->Tie( property_name.c_str(), this,
569  &FGTurbine::GetInjection, &FGTurbine::SetInjection);
570  property_name = base_property_name + "/seized";
571  PropertyManager->Tie( property_name.c_str(), &Seized);
572  property_name = base_property_name + "/stalled";
573  PropertyManager->Tie( property_name.c_str(), &Stalled);
574  property_name = base_property_name + "/bleed-factor";
575  PropertyManager->Tie( property_name.c_str(), this, &FGTurbine::GetBleedDemand, &FGTurbine::SetBleedDemand);
576  property_name = base_property_name + "/MaxN1";
