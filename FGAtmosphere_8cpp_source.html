102 double FGAtmosphere::ValidatePressure(double p, const string& msg, bool quiet) const
103 {
104  const double MinPressure = ConvertToPSF(1E-15, ePascals);
105  if (p < MinPressure) {
106  if (!quiet) {
107  cerr << msg << " " << p << " is too low." << endl
108  << msg << " is capped to " << MinPressure << endl;
109  }
110  return MinPressure;
111  }
112  return p;
113 }

116  // Make sure that the ambient temperature never drops to zero.
117  // According to Wikipedia, 1K is the temperature at the coolest natural place
118  // currently (2023) known in the Universe: the Boomerang Nebula
119 
120 double FGAtmosphere::ValidateTemperature(double t, const string& msg, bool quiet) const
121 {
122  // Minimum known temperature in the universe currently
123  constexpr double minUniverseTemperature = KelvinToRankine(1.0);
124 
125  if (t < minUniverseTemperature) {
126  if (!quiet) {
127  cerr << msg << " " << t << " is too low." << endl
128  << msg << " is capped to " << minUniverseTemperature << endl;
129  }
130  return minUniverseTemperature;
131  }
132  return t;
133 }

137 void FGAtmosphere::Calculate(double altitude)
138 {
139  FGPropertyNode* node = PropertyManager->GetNode();
140  double t =0.0;
141  if (!PropertyManager->HasNode("atmosphere/override/temperature"))
142  t = GetTemperature(altitude);
143  else
144  t = node->GetDouble("atmosphere/override/temperature");
145  Temperature = ValidateTemperature(t, "", true);
146 
147  double p = 0.0;
148  if (!PropertyManager->HasNode("atmosphere/override/pressure"))
149  p = GetPressure(altitude);
150  else
151  p = node->GetDouble("atmosphere/override/pressure");
152  Pressure = ValidatePressure(p, "", true);
153 
154  if (!PropertyManager->HasNode("atmosphere/override/density"))
155  Density = Pressure/(Reng*Temperature);
156  else
157  Density = node->GetDouble("atmosphere/override/density");
158 
159  Soundspeed = sqrt(SHRatio*Reng*Temperature);
160  PressureAltitude = CalculatePressureAltitude(Pressure, altitude);
161  DensityAltitude = CalculateDensityAltitude(Density, altitude);
162 
163  Viscosity = Beta * pow(Temperature, 1.5) / (SutherlandConstant + Temperature);
164  KinematicViscosity = Viscosity / Density;
165 }

208 double FGAtmosphere::ConvertToRankine(double t, eTemperature unit) const
209 {
210  double targetTemp=0; // in degrees Rankine
211 
212  switch(unit) {
213  case eFahrenheit:
214  targetTemp = t + 459.67;
215  break;
216  case eCelsius:
217  targetTemp = (t + 273.15) * 1.8;
218  break;
219  case eRankine:
220  targetTemp = t;
221  break;
222  case eKelvin:
223  targetTemp = t*1.8;
224  break;
225  default:
226  throw BaseException("Undefined temperature unit given");
227  }
228 
229  return targetTemp;
230 }

330 // The bitmasked value choices are as follows:
331 // unset: In this case (the default) JSBSim would only print
332 // out the normally expected messages, essentially echoing
333 // the config files as they are read. If the environment
334 // variable is not set, debug_lvl is set to 1 internally
335 // 0: This requests JSBSim not to output any messages
336 // whatsoever.
337 // 1: This value explicity requests the normal JSBSim
338 // startup messages
339 // 2: This value asks for a message to be printed out when
340 // a class is instantiated
341 // 4: When this value is set, a message is displayed when a
342 // FGModel object executes its Run() method
343 // 8: When this value is set, various runtime state variables
344 // are printed out periodically
345 // 16: When set various parameters are sanity checked and
346 // a message is printed out when they go out of bounds
347 
348 void FGAtmosphere::Debug(int from)
349 {
350  if (debug_lvl <= 0) return;
351 
352  if (debug_lvl & 1) { // Standard console startup message output
353  if (from == 0) { // Constructor
354  }
355  }
356  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
357  if (from == 0) std::cout << "Instantiated: FGAtmosphere" << std::endl;
358  if (from == 1) std::cout << "Destroyed: FGAtmosphere" << std::endl;
359  }
360  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
361  }
362  if (debug_lvl & 8 ) { // Runtime state variables
363  }
364  if (debug_lvl & 16) { // Sanity checking
365  }
366  if (debug_lvl & 128) { //
367  }
368  if (debug_lvl & 64) {
369  if (from == 0) { // Constructor
370  }
371  }
372 }
