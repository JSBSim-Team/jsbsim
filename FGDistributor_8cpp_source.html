53 FGDistributor::FGDistributor(FGFCS* fcs, Element* element)
54  : FGFCSComponent(fcs, element)
55 {
56  auto PropertyManager = fcs->GetPropertyManager();
57 
58  bind(element, PropertyManager.get()); // Bind() this component here in case it is used in its own
59  // definition for a sample-and-hold
60 
61  string type_string = element->GetAttributeValue("type");
62  if (type_string == "inclusive") Type = eInclusive;
63  else if (type_string == "exclusive") Type = eExclusive;
64  else {
65  throw("Not a known Distributor type, "+type_string);
66  }
67 
68  Element* case_element = element->FindElement("case");
69  while (case_element) {
70  Case* current_case = new Case;
71  Element* test_element = case_element->FindElement("test");
72  if (test_element) current_case->SetTest(new FGCondition(test_element, PropertyManager));
73  Element* prop_val_element = case_element->FindElement("property");
74  while (prop_val_element) {
75  string value_string = prop_val_element->GetAttributeValue("value");
76  string property_string = prop_val_element->GetDataLine();
77  current_case->AddPropValPair(new PropValPair(property_string, value_string,
78  PropertyManager, prop_val_element));
79  prop_val_element = case_element->FindNextElement("property");
80  }
81  Cases.push_back(current_case);
82  case_element = element->FindNextElement("case");
83  }
84 
85  Debug(0);
86 }

98 bool FGDistributor::Run(void )
99 {
100  bool completed = false;
101  for (auto Case: Cases) { // Loop through all Cases
102  if (Case->HasTest()) {
103  if (Case->GetTestResult() && !((Type == eExclusive) && completed)) {
104  Case->SetPropValPairs();
105  completed = true;
106  }
107  } else { // If no test present, execute always
108  Case->SetPropValPairs();
109  }
110  }
111 
112  return true;
113 }

116 // The bitmasked value choices are as follows:
117 // unset: In this case (the default) JSBSim would only print
118 // out the normally expected messages, essentially echoing
119 // the config files as they are read. If the environment
120 // variable is not set, debug_lvl is set to 1 internally
121 // 0: This requests JSBSim not to output any messages
122 // whatsoever.
123 // 1: This value explicity requests the normal JSBSim
124 // startup messages
125 // 2: This value asks for a message to be printed out when
126 // a class is instantiated
127 // 4: When this value is set, a message is displayed when a
128 // FGModel object executes its Run() method
129 // 8: When this value is set, various runtime state variables
130 // are printed out periodically
131 // 16: When set various parameters are sanity checked and
132 // a message is printed out when they go out of bounds

134 void FGDistributor::Debug(int from)
135 {
136  if (debug_lvl <= 0) return;
137 
138  if (debug_lvl & 1) { // Standard console startup message output
139  if (from == 0) { // Constructor
140  unsigned int ctr=0;
141  for (auto Case: Cases) {
142  std::cout << " Case: " << ctr << endl;
143  if (Case->HasTest()) {
144  Case->GetTest()->PrintCondition(" ");
145  } else {
146  std::cout << " Set these properties by default: " << std::endl;
147  }
148  std::cout << std::endl;
149  for (auto propVal = Case->IterPropValPairs(); propVal != Case->EndPropValPairs(); ++propVal) {
150  std::cout << " Set property " << (*propVal)->GetPropName();
151  if ((*propVal)->GetLateBoundProp()) std::cout << " (late bound)";
152  std::cout << " to " << (*propVal)->GetValString();
153  if ((*propVal)->GetLateBoundValue()) std::cout << " (late bound)";
154  std::cout << std::endl;
155  }
156  ctr++;
157  }
158  }
159  }
160  if (debug_lvl & 2 ) { // Instantiation/Destruction notification
161  if (from == 0) cout << "Instantiated: FGDistributor" << endl;
162  if (from == 1) cout << "Destroyed: FGDistributor" << endl;
163  }
164  if (debug_lvl & 4 ) { // Run() method entry print for FGModel-derived objects
165  }
166  if (debug_lvl & 8 ) { // Runtime state variables
167  }
168  if (debug_lvl & 16) { // Sanity checking
169  }
170  if (debug_lvl & 64) {
171  if (from == 0) { // Constructor
172  }
173  }
174 }
