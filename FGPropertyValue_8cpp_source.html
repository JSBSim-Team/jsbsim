42 FGPropertyValue::FGPropertyValue(const std::string& propName,
43  std::shared_ptr<FGPropertyManager> propertyManager, Element* el)
44  : PropertyManager(propertyManager), PropertyNode(nullptr), XML_def(el),
45  PropertyName(propName), Sign(1.0)
46 {
47  if (PropertyName[0] == '-') {
48  PropertyName.erase(0,1);
49  Sign = -1.0;
50  }
51 
52  if (PropertyManager->HasNode(PropertyName))
53  PropertyNode = PropertyManager->GetNode(PropertyName);
54 }
55

57 
58 FGPropertyNode* FGPropertyValue::GetNode(void) const
59 {
60  if (PropertyNode) return PropertyNode;
61 
62  // Manage late binding.
63  PropertyNode = PropertyManager->GetNode(PropertyName);
64  if (!PropertyNode) {
65  if (XML_def)
66  cerr << XML_def->ReadFrom()
67  << "Property " << PropertyName << " does not exist" << endl;
68  throw BaseException("FGPropertyValue::GetValue() The property " +
69  PropertyName + " does not exist.");
70  }
71 
72  XML_def = nullptr; // Now that the property is bound, we no longer need that.
73 
74  return PropertyNode;
75 }
76

79 double FGPropertyValue::GetValue(void) const
80 {
81  return GetNode()->getDoubleValue()*Sign;
82 }

86 void FGPropertyValue::SetValue(double value)
87 {
88  // SetValue() ignores the Sign flag. So make sure it is never called with a
89  // negative sign.
90  assert(Sign == 1);
91  GetNode()->setDoubleValue(value);
92 }
