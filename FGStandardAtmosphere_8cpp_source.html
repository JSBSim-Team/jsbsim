187 // Get the actual pressure as modeled at a specified altitude
188 // These calculations are from equations 33a and 33b in the U.S. Standard
189 // Atmosphere document referenced in the documentation for this code.
190 
191 double FGStandardAtmosphere::GetPressure(double altitude) const
192 {
193  double GeoPotAlt = GeopotentialAltitude(altitude);
194 
195  // Iterate through the altitudes to find the current Base Altitude
196  // in the table. That is, if the current altitude (the argument passed in)
197  // is 20000 ft, then the base altitude from the table is 0.0. If the
198  // passed-in altitude is 40000 ft, the base altitude is 36089.2388 ft (and
199  // the index "b" is 2 - the second entry in the table).
200  double BaseAlt = StdAtmosTemperatureTable(1,0);
201  unsigned int numRows = StdAtmosTemperatureTable.GetNumRows();
202  unsigned int b;
203 
204  for (b=0; b < numRows-2; ++b) {
205  double testAlt = StdAtmosTemperatureTable(b+2,0);
206  if (GeoPotAlt < testAlt)
207  break;
208  BaseAlt = testAlt;
209  }
210 
211  double Tmb = GetTemperature(GeometricAltitude(BaseAlt));
212  double deltaH = GeoPotAlt - BaseAlt;
213  double Lmb = LapseRates[b];
214 
215  if (Lmb != 0.0) {
216  double Exp = g0 / (Rdry*Lmb);
217  double factor = Tmb/(Tmb + Lmb*deltaH);
218  return PressureBreakpoints[b]*pow(factor, Exp);
219  } else
220  return PressureBreakpoints[b]*exp(-g0*deltaH/(Rdry*Tmb));
221 }

242 // Get the modeled temperature at a specified altitude, including any bias or gradient
243 // effects.
244 
245 double FGStandardAtmosphere::GetTemperature(double altitude) const
246 {
247  double GeoPotAlt = GeopotentialAltitude(altitude);
248 
249  double T;
250 
251  if (GeoPotAlt >= 0.0) {
252  T = StdAtmosTemperatureTable.GetValue(GeoPotAlt);
253 
254  if (GeoPotAlt <= GradientFadeoutAltitude)
255  T -= TemperatureDeltaGradient * GeoPotAlt;
256  }
257  else {
258  // We don't need to add TemperatureDeltaGradient*GeoPotAlt here because
259  // the lapse rate vector already accounts for the temperature gradient.
260  T = StdAtmosTemperatureTable.GetValue(0.0) + GeoPotAlt*LapseRates[0];
261  }
262 
263  T += TemperatureBias;
264 
265  if (GeoPotAlt <= GradientFadeoutAltitude)
266  T += TemperatureDeltaGradient * GradientFadeoutAltitude;
267 
268  return T;
269 }

273 
274 double FGStandardAtmosphere::GetStdTemperature(double altitude) const
275 {
276  double GeoPotAlt = GeopotentialAltitude(altitude);
277 
278  if (GeoPotAlt >= 0.0)
279  return StdAtmosTemperatureTable.GetValue(GeoPotAlt);
280  else
281  return StdAtmosTemperatureTable.GetValue(0.0) + GeoPotAlt*LapseRates[0];
282 }

328 void FGStandardAtmosphere::SetTemperature(double t, double h, eTemperature unit)
329 {
330  double targetTemp = ConvertToRankine(t, unit);
331  double GeoPotAlt = GeopotentialAltitude(h);
332  double bias = targetTemp - GetStdTemperature(h);
333 
334  if (GeoPotAlt <= GradientFadeoutAltitude)
335  bias -= TemperatureDeltaGradient * (GradientFadeoutAltitude - GeoPotAlt);
336 
337  SetTemperatureBias(eRankine, bias);
338  CalculatePressureBreakpoints(SLpressure);
339 
340  SLtemperature = GetTemperature(0.0);
341  CalculateSLSoundSpeedAndDensity();
342 }

346 void FGStandardAtmosphere::SetTemperatureBias(eTemperature unit, double t)
347 {
348  // Retrieve the minimum temperature in the standard atmosphere, may not be the
349  // last row in future if for example it's extended and maybe there is some
350  // temperature inversion layer etc. So run through and find the minimum.
351  const double minStdAtmosphereTemp = StdAtmosTemperatureTable.GetMinValue();
352 
353  // Minimum known temperature in the universe currently
354  constexpr double minUniverseTemperature = KelvinToRankine(1.0);
355 
356  if (unit == eCelsius || unit == eKelvin)
357  t *= 1.80; // If temp delta "t" is given in metric, scale up to English
358 
359  TemperatureBias = t;
360  // Confirm the temperature bias isn't going to result in an atmosphere
361  // temperature lower than the lowest known temperature in the universe
362  if (minStdAtmosphereTemp + TemperatureBias < minUniverseTemperature) {
363  double minBias = minUniverseTemperature - minStdAtmosphereTemp;
364  cerr << "The temperature bias " << TemperatureBias << " R is too low. "
365  << "It could result in temperatures below the absolute zero." << endl
366  << "Temperature bias is therefore capped to " << minBias << endl;
367  TemperatureBias = minBias;
368  }
369 
370  CalculatePressureBreakpoints(SLpressure);
371 
372  SLtemperature = GetTemperature(0.0);
373  CalculateSLSoundSpeedAndDensity();
374 }

395 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
396 // Sets a temperature delta at the supplied altitude that is ramped out by 86 km.
397 // After this calculation is performed, the lapse rates and pressure breakpoints
398 // must be recalculated. Since we are calculating a delta here and not an actual
399 // temperature, we only need to be concerned about a scale factor and not
400 // the actual temperature itself.
401 
402 void FGStandardAtmosphere::SetTemperatureGradedDelta(double deltemp, double h, eTemperature unit)
403 {
404  // Retrieve the minimum temperature in the standard atmosphere, may not be the
405  // last row in future if for example it's extended and maybe there is some
406  // temperature inversion layer etc. So run through and find the minimum.
407  const double minStdAtmosphereTemp = StdAtmosTemperatureTable.GetMinValue();
408  const double minDeltaTemperature = minStdAtmosphereTemp - StdSLtemperature;
409 
410  if (unit == eCelsius || unit == eKelvin)
411  deltemp *= 1.80; // If temp delta "t" is given in metric, scale up to English
412 
413  if (deltemp <= minDeltaTemperature) {
414  cerr << "The temperature delta " << deltemp << " R is too low. "
415  << "It could result in temperatures below the absolute zero." << endl
416  << "Temperature delta is therefore capped to " << minDeltaTemperature << endl;
417  deltemp = minDeltaTemperature;
418  }
419 
420  TemperatureDeltaGradient = deltemp/(GradientFadeoutAltitude - GeopotentialAltitude(h));
421  CalculateLapseRates();
422  CalculatePressureBreakpoints(SLpressure);
423 
424  SLtemperature = GetTemperature(0.0);
425  CalculateSLSoundSpeedAndDensity();
426 }

447 //%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
448 // This function calculates (or recalculates) the lapse rate over an altitude range
449 // where the "bh" in this case refers to the index of the base height in the
450 // StdAtmosTemperatureTable table. This function should be called anytime the
451 // temperature table is altered, such as when a gradient is applied across the
452 // temperature table for a range of altitudes.
453 
454 void FGStandardAtmosphere::CalculateLapseRates()
455 {
456  unsigned int numRows = StdAtmosTemperatureTable.GetNumRows();
457  LapseRates.clear();
458 
459  for (unsigned int bh=0; bh < numRows-1; bh++)
460  {
461  double t0 = StdAtmosTemperatureTable(bh+1,1);
462  double t1 = StdAtmosTemperatureTable(bh+2,1);
463  double h0 = StdAtmosTemperatureTable(bh+1,0);
464  double h1 = StdAtmosTemperatureTable(bh+2,0);
465  LapseRates.push_back((t1 - t0) / (h1 - h0) - TemperatureDeltaGradient);
466  }
467 }

471 void FGStandardAtmosphere::CalculatePressureBreakpoints(double SLpress)
472 {
473  PressureBreakpoints[0] = SLpress;
474 
475  for (unsigned int b=0; b<PressureBreakpoints.size()-1; b++) {
476  double BaseTemp = StdAtmosTemperatureTable(b+1,1);
477  double BaseAlt = StdAtmosTemperatureTable(b+1,0);
478  double UpperAlt = StdAtmosTemperatureTable(b+2,0);
479  double deltaH = UpperAlt - BaseAlt;
480  double Tmb = BaseTemp
481  + TemperatureBias
482  + (GradientFadeoutAltitude - BaseAlt)*TemperatureDeltaGradient;
483  if (LapseRates[b] != 0.00) {
484  double Lmb = LapseRates[b];
485  double Exp = g0 / (Rdry*Lmb);
486  double factor = Tmb/(Tmb + Lmb*deltaH);
487  PressureBreakpoints[b+1] = PressureBreakpoints[b]*pow(factor, Exp);
488  } else {
489  PressureBreakpoints[b+1] = PressureBreakpoints[b]*exp(-g0*deltaH/(Rdry*Tmb));
490  }
491  }
492 }

526 double FGStandardAtmosphere::CalculateDensityAltitude(double density, double geometricAlt)
527 {
528  // Work out which layer we're dealing with
529  unsigned int b = 0;
530  for (; b < StdDensityBreakpoints.size() - 2; b++) {
531  if (density >= StdDensityBreakpoints[b + 1])
532  break;
533  }
534 
535  // Get layer properties
536  double Tmb = StdAtmosTemperatureTable(b + 1, 1);
537  double Hb = StdAtmosTemperatureTable(b + 1, 0);
538  double Lmb = StdLapseRates[b];
539  double pb = StdDensityBreakpoints[b];
540 
541  double density_altitude = 0.0;
542 
543  // https://en.wikipedia.org/wiki/Barometric_formula for density solved for H
544  if (Lmb != 0.0) {
545  double Exp = -1.0 / (1.0 + g0/(Rdry*Lmb));
546  density_altitude = Hb + (Tmb / Lmb) * (pow(density / pb, Exp) - 1);
547  } else {
548  double Factor = -Rdry*Tmb / g0;
549  density_altitude = Hb + Factor * log(density / pb);
550  }
551 
552  return GeometricAltitude(density_altitude);
553 }

557 double FGStandardAtmosphere::CalculatePressureAltitude(double pressure, double geometricAlt)
558 {
559  // Work out which layer we're dealing with
560  unsigned int b = 0;
561  for (; b < StdPressureBreakpoints.size() - 2; b++) {
562  if (pressure >= StdPressureBreakpoints[b + 1])
563  break;
564  }
565 
566  // Get layer properties
567  double Tmb = StdAtmosTemperatureTable(b + 1, 1);
568  double Hb = StdAtmosTemperatureTable(b + 1, 0);
569  double Lmb = StdLapseRates[b];
570  double Pb = StdPressureBreakpoints[b];
571 
572  double pressure_altitude = 0.0;
573 
574  if (Lmb != 0.00) {
575  // Equation 33(a) from ISA document solved for H
576  double Exp = -Rdry*Lmb / g0;
577  pressure_altitude = Hb + (Tmb / Lmb) * (pow(pressure / Pb, Exp) - 1);
578  } else {
579  // Equation 33(b) from ISA document solved for H
580  double Factor = -Rdry*Tmb / g0;
581  pressure_altitude = Hb + Factor * log(pressure / Pb);
582  }
583 
584  return GeometricAltitude(pressure_altitude);
585 }

597 void FGStandardAtmosphere::ValidateVaporMassFraction(double h)
598 {
599  if (SaturatedVaporPressure < Pressure) {
600  double VaporPressure = Pressure*VaporMassFraction / (VaporMassFraction+Rdry/Rwater);
601  if (VaporPressure > SaturatedVaporPressure)
602  VaporMassFraction = Rdry * SaturatedVaporPressure / (Rwater * (Pressure - SaturatedVaporPressure));
603  }
604 
605  double GeoPotAlt = GeopotentialAltitude(h);
606  double maxFraction = 1E-6*MaxVaporMassFraction.GetValue(GeoPotAlt);
607 
608  if ((VaporMassFraction > maxFraction) || (VaporMassFraction < 0.0))
609  VaporMassFraction = maxFraction;
610 
611  // Update the gas constant factor
612  Reng = (VaporMassFraction*Rwater + Rdry)/(1.0 + VaporMassFraction);
613 }

617 void FGStandardAtmosphere::SetDewPoint(eTemperature unit, double dewpoint)
618 {
619  double dewPoint_R = ConvertToRankine(dewpoint, unit);
620  constexpr double minDewPoint = -CelsiusToRankine(c) + 1.0;
621 
622  if (dewPoint_R <= minDewPoint) {
623  cerr << "The dew point temperature " << dewPoint_R << " is lower than "
624  << minDewPoint << " R." << endl
625  << "Dew point is therefore capped to " << minDewPoint << endl;

627  }

641 double FGStandardAtmosphere::GetDewPoint(eTemperature to) const
642 {
643  double dewpoint_degC;
644  double VaporPressure = Pressure*VaporMassFraction / (VaporMassFraction+Rdry/Rwater);
645 
646  if (VaporPressure <= 0.0)
647  dewpoint_degC = -c;
648  else {
649  double x = log(VaporPressure/a);
650  dewpoint_degC = c*x / (b - x);
651  }
652 
653  return ConvertFromRankine(CelsiusToRankine(dewpoint_degC), to);
654 }

658 void FGStandardAtmosphere::SetVaporPressure(ePressure unit, double Pa)
659 {
660  double altitude = CalculatePressureAltitude(Pressure, 0.0);
661  double VaporPressure = ConvertToPSF(Pa, unit);
662  if (VaporPressure < 0.0) {
663  cerr << "The vapor pressure cannot be negative." << endl
664  << "Vapor pressure is set to 0.0" << endl;
665  VaporPressure = 0.0;
666  } else if (VaporPressure >= Pressure) {
667  cerr << "The vapor pressure " << VaporPressure
668  << " PSF is higher than the ambient pressure." << endl
669  << "Vapor pressure is therefore capped to " << Pressure-1.0 << endl;
670  VaporPressure = Pressure - 1.0;
671  }
672  VaporMassFraction = Rdry * VaporPressure / (Rwater * (Pressure - VaporPressure));
673  ValidateVaporMassFraction(altitude);
674 }
