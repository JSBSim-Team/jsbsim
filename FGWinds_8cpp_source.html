140 bool FGWinds::Run(bool Holding)
141 {
142  if (FGModel::Run(Holding)) return true;
143  if (Holding) return false;
144 
145  if (turbType != ttNone) Turbulence(in.AltitudeASL);
146  if (oneMinusCosineGust.gustProfile.Running) CosineGust();
147 
148  vTotalWindNED = vWindNED + vGustNED + vCosineGust + vTurbulenceNED;
149 
150  // psiw (Wind heading) is the direction the wind is blowing towards
151  if (vWindNED(eX) != 0.0) psiw = atan2( vWindNED(eY), vWindNED(eX) );
152  if (psiw < 0) psiw += 2*M_PI;
153 
154  Debug(2);
155  return false;
156 }

194 void FGWinds::Turbulence(double h)
195 {
196  switch (turbType) {
197 
198  case ttCulp: {
199 
200  vTurbPQR(eP) = wind_from_clockwise;
201  if (TurbGain == 0.0) return;
202 
203  // keep the inputs within allowable limts for this model
204  if (TurbGain < 0.0) TurbGain = 0.0;
205  if (TurbGain > 1.0) TurbGain = 1.0;
206  if (TurbRate < 0.0) TurbRate = 0.0;
207  if (TurbRate > 30.0) TurbRate = 30.0;
208  if (Rhythmicity < 0.0) Rhythmicity = 0.0;
209  if (Rhythmicity > 1.0) Rhythmicity = 1.0;
210 
211  // generate a sine wave corresponding to turbulence rate in hertz
212  double time = FDMExec->GetSimTime();
213  double sinewave = sin( time * TurbRate * 6.283185307 );
214 
215  double random = 0.0;
216  if (target_time == 0.0) {
217  strength = random = generator->GetUniformRandomNumber();
218  target_time = time + 0.71 + (random * 0.5);
219  }
220  if (time > target_time) {
221  spike = 1.0;
222  target_time = 0.0;
223  }
224 
225  // max vertical wind speed in fps, corresponds to TurbGain = 1.0
226  double max_vs = 40;
227 
228  vTurbulenceNED.InitMatrix();
229  double delta = strength * max_vs * TurbGain * (1-Rhythmicity) * spike;
230 
231  // Vertical component of turbulence.
232  vTurbulenceNED(eDown) = sinewave * max_vs * TurbGain * Rhythmicity;
233  vTurbulenceNED(eDown)+= delta;
234  if (in.DistanceAGL/in.wingspan < 3.0)
235  vTurbulenceNED(eDown) *= in.DistanceAGL/in.wingspan * 0.3333;
236 
237  // Yaw component of turbulence.
238  vTurbulenceNED(eNorth) = sin( delta * 3.0 );
239  vTurbulenceNED(eEast) = cos( delta * 3.0 );
240 
241  // Roll component of turbulence. Clockwise vortex causes left roll.
242  vTurbPQR(eP) += delta * 0.04;
243 
244  spike = spike * 0.9;
245  break;
246  }
247  case ttMilspec:
248  case ttTustin: {
249 
250  // an index of zero means turbulence is disabled
251  // airspeed occurs as divisor in the code below
252  if (probability_of_exceedence_index == 0 || in.V == 0) {
253  vTurbulenceNED(eNorth) = vTurbulenceNED(eEast) = vTurbulenceNED(eDown) = 0.0;
254  vTurbPQR(eP) = vTurbPQR(eQ) = vTurbPQR(eR) = 0.0;
255  return;
256  }
257 
258  // Turbulence model according to MIL-F-8785C (Flying Qualities of Piloted Aircraft)
259  double b_w = in.wingspan, L_u, L_w, sig_u, sig_w;
260 
261  if (b_w == 0.) b_w = 30.;
262 
263  // clip height functions at 10 ft
264  if (h <= 10.) h = 10;
265 
266  // Scale lengths L and amplitudes sigma as function of height
267  if (h <= 1000) {
268  L_u = h/pow(0.177 + 0.000823*h, 1.2); // MIL-F-8785c, Fig. 10, p. 55
269  L_w = h;
270  sig_w = 0.1*windspeed_at_20ft;
271  sig_u = sig_w/pow(0.177 + 0.000823*h, 0.4); // MIL-F-8785c, Fig. 11, p. 56
272  } else if (h <= 2000) {
273  // linear interpolation between low altitude and high altitude models
274  L_u = L_w = 1000 + (h-1000.)/1000.*750.;
275  sig_u = sig_w = 0.1*windspeed_at_20ft
276  + (h-1000.)/1000.*(POE_Table->GetValue(probability_of_exceedence_index, h) - 0.1*windspeed_at_20ft);
277  } else {
278  L_u = L_w = 1750.; // MIL-F-8785c, Sec. 3.7.2.1, p. 48
279  sig_u = sig_w = POE_Table->GetValue(probability_of_exceedence_index, h);
280  }
281 
282  // keep values from last timesteps
283  // TODO maybe use deque?
284  static double
285  xi_u_km1 = 0, nu_u_km1 = 0,
286  xi_v_km1 = 0, xi_v_km2 = 0, nu_v_km1 = 0, nu_v_km2 = 0,
287  xi_w_km1 = 0, xi_w_km2 = 0, nu_w_km1 = 0, nu_w_km2 = 0,
288  xi_p_km1 = 0, nu_p_km1 = 0,
289  xi_q_km1 = 0, xi_r_km1 = 0;
290 
291 
292  double
293  T_V = in.totalDeltaT, // for compatibility of nomenclature
294  sig_p = 1.9/sqrt(L_w*b_w)*sig_w, // Yeager1998, eq. (8)
295  //sig_q = sqrt(M_PI/2/L_w/b_w), // eq. (14)
296  //sig_r = sqrt(2*M_PI/3/L_w/b_w), // eq. (17)
297  L_p = sqrt(L_w*b_w)/2.6, // eq. (10)
298  tau_u = L_u/in.V, // eq. (6)
299  tau_w = L_w/in.V, // eq. (3)
300  tau_p = L_p/in.V, // eq. (9)
301  tau_q = 4*b_w/M_PI/in.V, // eq. (13)
302  tau_r =3*b_w/M_PI/in.V, // eq. (17)
303  nu_u = generator->GetNormalRandomNumber(),
304  nu_v = generator->GetNormalRandomNumber(),
305  nu_w = generator->GetNormalRandomNumber(),
306  nu_p = generator->GetNormalRandomNumber(),
307  xi_u=0, xi_v=0, xi_w=0, xi_p=0, xi_q=0, xi_r=0;
308 
309  // values of turbulence NED velocities
310 
311  if (turbType == ttTustin) {
312  // the following is the Tustin formulation of Yeager's report
313  double
314  omega_w = in.V/L_w, // hidden in nomenclature p. 3
315  omega_v = in.V/L_u, // this is defined nowhere
316  C_BL = 1/tau_u/tan(T_V/2/tau_u), // eq. (19)
317  C_BLp = 1/tau_p/tan(T_V/2/tau_p), // eq. (22)
318  C_BLq = 1/tau_q/tan(T_V/2/tau_q), // eq. (24)
319  C_BLr = 1/tau_r/tan(T_V/2/tau_r); // eq. (26)
320 
321  // all values calculated so far are strictly positive, except for
322  // the random numbers nu_*. This means that in the code below, all
323  // divisors are strictly positive, too, and no floating point
324  // exception should occur.
325  xi_u = -(1 - C_BL*tau_u)/(1 + C_BL*tau_u)*xi_u_km1
326  + sig_u*sqrt(2*tau_u/T_V)/(1 + C_BL*tau_u)*(nu_u + nu_u_km1); // eq. (18)
327  xi_v = -2*(sqr(omega_v) - sqr(C_BL))/sqr(omega_v + C_BL)*xi_v_km1
328  - sqr(omega_v - C_BL)/sqr(omega_v + C_BL) * xi_v_km2
329  + sig_u*sqrt(3*omega_v/T_V)/sqr(omega_v + C_BL)*(
330  (C_BL + omega_v/sqrt(3.))*nu_v
331  + 2/sqrt(3.)*omega_v*nu_v_km1
332  + (omega_v/sqrt(3.) - C_BL)*nu_v_km2); // eq. (20) for v
333  xi_w = -2*(sqr(omega_w) - sqr(C_BL))/sqr(omega_w + C_BL)*xi_w_km1
334  - sqr(omega_w - C_BL)/sqr(omega_w + C_BL) * xi_w_km2
335  + sig_w*sqrt(3*omega_w/T_V)/sqr(omega_w + C_BL)*(
336  (C_BL + omega_w/sqrt(3.))*nu_w
337  + 2/sqrt(3.)*omega_w*nu_w_km1
338  + (omega_w/sqrt(3.) - C_BL)*nu_w_km2); // eq. (20) for w
339  xi_p = -(1 - C_BLp*tau_p)/(1 + C_BLp*tau_p)*xi_p_km1
340  + sig_p*sqrt(2*tau_p/T_V)/(1 + C_BLp*tau_p) * (nu_p + nu_p_km1); // eq. (21)
341  xi_q = -(1 - 4*b_w*C_BLq/M_PI/in.V)/(1 + 4*b_w*C_BLq/M_PI/in.V) * xi_q_km1
342  + C_BLq/in.V/(1 + 4*b_w*C_BLq/M_PI/in.V) * (xi_w - xi_w_km1); // eq. (23)
343  xi_r = - (1 - 3*b_w*C_BLr/M_PI/in.V)/(1 + 3*b_w*C_BLr/M_PI/in.V) * xi_r_km1
344  + C_BLr/in.V/(1 + 3*b_w*C_BLr/M_PI/in.V) * (xi_v - xi_v_km1); // eq. (25)
345 
346  } else if (turbType == ttMilspec) {
347  // the following is the MIL-STD-1797A formulation
348  // as cited in Yeager's report
349  xi_u = (1 - T_V/tau_u) *xi_u_km1 + sig_u*sqrt(2*T_V/tau_u)*nu_u; // eq. (30)
350  xi_v = (1 - 2*T_V/tau_u)*xi_v_km1 + sig_u*sqrt(4*T_V/tau_u)*nu_v; // eq. (31)
351  xi_w = (1 - 2*T_V/tau_w)*xi_w_km1 + sig_w*sqrt(4*T_V/tau_w)*nu_w; // eq. (32)
352  xi_p = (1 - T_V/tau_p) *xi_p_km1 + sig_p*sqrt(2*T_V/tau_p)*nu_p; // eq. (33)
353  xi_q = (1 - T_V/tau_q) *xi_q_km1 + M_PI/4/b_w*(xi_w - xi_w_km1); // eq. (34)
354  xi_r = (1 - T_V/tau_r) *xi_r_km1 + M_PI/3/b_w*(xi_v - xi_v_km1); // eq. (35)
355  }
356 
357  // rotate by wind azimuth and assign the velocities
358  double cospsi = cos(psiw), sinpsi = sin(psiw);
359  vTurbulenceNED(eNorth) = cospsi*xi_u + sinpsi*xi_v;
360  vTurbulenceNED(eEast) = -sinpsi*xi_u + cospsi*xi_v;
361  vTurbulenceNED(eDown) = xi_w;
362 
363  vTurbPQR(eP) = cospsi*xi_p + sinpsi*xi_q;
364  vTurbPQR(eQ) = -sinpsi*xi_p + cospsi*xi_q;
365  vTurbPQR(eR) = xi_r;
366 
367  // vTurbPQR is in the body fixed frame, not NED
368  vTurbPQR = in.Tl2b*vTurbPQR;
369 
370  // hand on the values for the next timestep
371  xi_u_km1 = xi_u; nu_u_km1 = nu_u;
372  xi_v_km2 = xi_v_km1; xi_v_km1 = xi_v; nu_v_km2 = nu_v_km1; nu_v_km1 = nu_v;
373  xi_w_km2 = xi_w_km1; xi_w_km1 = xi_w; nu_w_km2 = nu_w_km1; nu_w_km1 = nu_w;
374  xi_p_km1 = xi_p; nu_p_km1 = nu_p;
375  xi_q_km1 = xi_q;
376  xi_r_km1 = xi_r;
377 
378  }
379  default:
380  break;
381  }
382 
383  TurbDirection = atan2( vTurbulenceNED(eEast), vTurbulenceNED(eNorth))*radtodeg;
384 
385 }

389 double FGWinds::CosineGustProfile(double startDuration, double steadyDuration, double endDuration, double elapsedTime)
390 {
391  double factor = 0.0;
392  if (elapsedTime >= 0 && elapsedTime <= startDuration) {
393  factor = (1.0 - cos(M_PI*elapsedTime/startDuration))/2.0;
394  } else if (elapsedTime > startDuration && (elapsedTime <= (startDuration + steadyDuration))) {
395  factor = 1.0;
396  } else if (elapsedTime > (startDuration + steadyDuration) && elapsedTime <= (startDuration + steadyDuration + endDuration)) {
397  factor = (1-cos(M_PI*(1-(elapsedTime-(startDuration + steadyDuration))/endDuration)))/2.0;
398  } else {
399  factor = 0.0;
400  }
401 
402  return factor;
403 }

407 void FGWinds::CosineGust()
408 {
409  struct OneMinusCosineProfile& profile = oneMinusCosineGust.gustProfile;
410 
411  double factor = CosineGustProfile( profile.startupDuration,
412  profile.steadyDuration,
413  profile.endDuration,
414  profile.elapsedTime);
415  // Normalize the gust wind vector
416  oneMinusCosineGust.vWind.Normalize();
417 
418  if (oneMinusCosineGust.vWindTransformed.Magnitude() == 0.0) {
419  switch (oneMinusCosineGust.gustFrame) {
420  case gfBody:
421  oneMinusCosineGust.vWindTransformed = in.Tl2b.Inverse() * oneMinusCosineGust.vWind;
422  break;
423  case gfWind:
424  oneMinusCosineGust.vWindTransformed = in.Tl2b.Inverse() * in.Tw2b * oneMinusCosineGust.vWind;
425  break;
426  case gfLocal:
427  // this is the native frame - and the default.
428  oneMinusCosineGust.vWindTransformed = oneMinusCosineGust.vWind;
429  break;
430  default:
431  break;
432  }
433  }
434 
435  vCosineGust = factor * oneMinusCosineGust.vWindTransformed * oneMinusCosineGust.magnitude;
436 
437  profile.elapsedTime += in.totalDeltaT;
438 
439  if (profile.elapsedTime > (profile.startupDuration + profile.steadyDuration + profile.endDuration)) {
440  profile.Running = false;
441  profile.elapsedTime = 0.0;
442  oneMinusCosineGust.vWindTransformed.InitMatrix();
443  vCosineGust.InitMatrix(0);
444  }
445 }

459 // Calculates the distance between a specified point (where presumably the
460 // Up/Downburst is centered) and the current vehicle location. The distance
461 // here is calculated from the Haversine formula.
462 
463 double FGWinds::DistanceFromRingCenter(double lat, double lon)
464 {
465  double deltaLat = in.latitude - lat;
466  double deltaLong = in.longitude - lon;
467  double dLat2 = deltaLat/2.0;
468  double dLong2 = deltaLong/2.0;
469  double a = sin(dLat2)*sin(dLat2)
470  + cos(lat)*cos(in.latitude)*sin(dLong2)*sin(dLong2);
471  double c = 2.0*atan2(sqrt(a), sqrt(1.0 - a));
472  double d = in.planetRadius*c;
473  return d;
474 }

489 void FGWinds::bind(void)
490 {
491  typedef double (FGWinds::*PMF)(int) const;
492  typedef int (FGWinds::*PMFt)(void) const;
493  typedef void (FGWinds::*PMFd)(int,double);
494  typedef void (FGWinds::*PMFi)(int);
495  typedef double (FGWinds::*Ptr)(void) const;
496 
497  // User-specified steady, constant, wind properties (local navigational/geographic frame: N-E-D)
498  PropertyManager->Tie("atmosphere/psiw-rad", this, &FGWinds::GetWindPsi, &FGWinds::SetWindPsi);
499  PropertyManager->Tie("atmosphere/wind-north-fps", this, eNorth, (PMF)&FGWinds::GetWindNED,
500  (PMFd)&FGWinds::SetWindNED);
501  PropertyManager->Tie("atmosphere/wind-east-fps", this, eEast, (PMF)&FGWinds::GetWindNED,
502  (PMFd)&FGWinds::SetWindNED);
503  PropertyManager->Tie("atmosphere/wind-down-fps", this, eDown, (PMF)&FGWinds::GetWindNED,
504  (PMFd)&FGWinds::SetWindNED);
505  PropertyManager->Tie("atmosphere/wind-mag-fps", this, &FGWinds::GetWindspeed,
506  &FGWinds::SetWindspeed);
507

581 // 4: When this value is set, a message is displayed when a
582 // FGModel object executes its Run() method
583 // 8: When this value is set, various runtime state variables
584 // are printed out periodically
585 // 16: When set various parameters are sanity checked and
586 // a message is printed out when they go out of bounds

605  }
